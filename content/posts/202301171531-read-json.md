function readJson(fileName) {
  var file = new File(dataPath + "/" + fileName + ".txt");
  var currentLine;
  var jsonStuff = [];
  file.open("r");
  while (!file.eof) {
    currentLine = file.readln();
    jsonStuff.push(currentLine);
  }
  file.close();
  jsonStuff = jsonStuff.join("");
  var parsedJson = JSON.parse(jsonStuff);
  return parsedJson;
}
